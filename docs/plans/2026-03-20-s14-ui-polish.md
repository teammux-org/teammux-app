### Task 2: Create PaneIconRail.swift

**Files:**
- Create: `macos/Sources/Teammux/RightPane/PaneIconRail.swift`

**Step 1: Write PaneIconRail.swift**

```swift
import SwiftUI

// MARK: - PaneIconRail

/// Vertical scrollable icon rail for right pane navigation.
///
/// Sits on the far right edge of the right pane. Each icon is an
/// SF Symbol with a 44pt tap target, tooltip on hover, and
/// active/inactive visual states.
struct PaneIconRail: View {
    @Binding var activeTab: RightTab

    var body: some View {
        ScrollView(.vertical, showsIndicators: false) {
            VStack(spacing: 4) {
                ForEach(RightTab.allCases) { tab in
                    iconButton(for: tab)
                }
            }
            .padding(.vertical, 8)
        }
        .frame(width: 44)
        .background(Color(nsColor: .controlBackgroundColor))
    }

    // MARK: - Icon button

    private func iconButton(for tab: RightTab) -> some View {
        Button(action: {
            activeTab = tab
        }) {
            Image(systemName: tab.iconName)
                .font(.system(size: 16))
                .foregroundColor(activeTab == tab ? .accentColor : .secondary)
                .frame(width: 36, height: 36)
                .background(
                    RoundedRectangle(cornerRadius: 8)
                        .fill(activeTab == tab ? Color.accentColor.opacity(0.15) : Color.clear)
                )
        }
        .buttonStyle(.plain)
        .help(tab.rawValue)
        .frame(width: 44, height: 44)
        .contentShape(Rectangle())
    }
}
```

### Task 7: Keyboard shortcuts — Cmd+1..7 pane switching

**Files:**
- Modify: `macos/Sources/Teammux/RightPane/RightPaneView.swift`

**Step 1: Add keyboard shortcut index to RightTab**

Add a computed property to `RightTab` for the Cmd+N shortcut:

```swift
/// Keyboard shortcut number (1-based) for Cmd+N switching.
var shortcutIndex: Int {
    switch self {
    case .teamLead: return 1
    case .git:      return 2
    case .diff:     return 3
    case .liveFeed: return 4
    case .dispatch: return 5
    case .context:  return 6
    case .you:      return 7
    }
}

/// Returns the RightTab for a given shortcut number, or nil.
static func fromShortcut(_ number: Int) -> RightTab? {
    allCases.first { $0.shortcutIndex == number }
}
```

**Step 2: Add keyboard event handler to RightPaneView body**

Add `.onKeyPress` to the body's HStack for Cmd+1..7:

Note: `.onKeyPress` requires macOS 14+. Since we target macOS 15 Sequoia, this is safe.

```swift
var body: some View {
    HStack(spacing: 0) {
        tabContent
            .frame(maxWidth: .infinity, maxHeight: .infinity)
            .animation(.easeInOut(duration: 0.15), value: activeTab)

        Divider()

        PaneIconRail(activeTab: $activeTab)
    }
    .background(Color(nsColor: .controlBackgroundColor))
    .onKeyPress(keys: Set("1234567".map { KeyEquivalent(Character(String($0))) }), phases: .down) { press in
        guard press.modifiers == .command,
              let number = Int(String(press.key.character)),
              let tab = RightTab.fromShortcut(number) else {
            return .ignored
        }
        activeTab = tab
        return .handled
    }
}
```

Also note the `.animation(.easeInOut(duration: 0.15), value: activeTab)` on the content — this gives a subtle cross-fade on pane switch.
